Create:
    md5sum *.csv > allclean.md5
    tar cf allclean.tar allclean
    xz -z -k -9e -T 0 allclean.tar
    base64 -w 0 allclean.tar.xz > allclean.txz.b64
    rm allclean.tar
    mkdir allclean_b64_split
    cd allclean_b64_split
    split -a 3 -b 10M -d ../allclean.txz.b64 allclean.tx64. & \
        pv -d $(pgrep split):0
    ls > index.html
    sed -i -ne '/index/ d; s/.*/<a href="&"><\/a>/; p' index.html

Push to github

Download:
# Proble for the downloaded marker file
if [ ! -e dl.ok ]; then
    # Gets the index of files to download and passes it on
    ( (wget -nv -O - --show-progress \
        https://raw.githubusercontent.com/xxc3nsoredxx/dissconnect_data/2afe781db2d2a17b9efab0c50294f8ffb554f6ac/allclean_b64_split/index.html | \
        # Downloads the required files and passes them on as a constant(ish)
        # stream of data
        wget -nv -i - --force-html \
        -B https://raw.githubusercontent.com/xxc3nsoredxx/dissconnect_data/ad4dfb51adf657d23e173f4a28a9b740b83c2841/allclean_b64_split/ \
        -O - --show-progress | \
        # Once enough data has stramed in:
        #   * Undoes the base64 encoding and passes it on to tar
        #   * tar both un-xz's and un-tars one dataset
        base64 -d -w 0 | tar xJ --strip=1) && \
        # Once all the datasets are downloaded, it compares the downloaded md5
        # hashes for each, if they mach the marker file is created
        md5sum -c allclean.md5 && touch dl.ok) || \
    (echo "Error downloading files, try again later"; rm *all*; exit 1)

    # If the dowload was a success, create the merged csv file used later on
    if [ ! -e *merged.csv ]
        cat {*1.,*2*.}csv > output_allclean_merged.csv
    fi
fi
